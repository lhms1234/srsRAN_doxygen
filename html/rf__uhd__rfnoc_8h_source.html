334 
335  // DUC -> Radio
336  UHD_LOG_DEBUG(graph->get_name(), "Connecting " << duc_id[i] << " -> " << radio_id[i]);

338

344  "Connecting " << fifo_id[0] << ":" << dma_port << " -> " << duc_id[i] << ":" << j);
345  graph->connect(fifo_id[0], dma_port, duc_id[i], j, nof_samples_per_packet);
346  }
347  } else {
348  // FIFO -> DUC
349  for (size_t j = 0; j < nof_channels; j++) {
350  size_t fifo_idx = i * nof_channels + j;

409  uhd_error usrp_make(const uhd::device_addr_t& _device_addr, uint32_t nof_channels_) override
410  {
411  // Copy device address
412  uhd::device_addr_t device_addr = _device_addr;
413 
414  // Parse arguments, removes
415  uhd_error err = parse_args(device_addr);
416  if (err != UHD_ERROR_NONE) {
417  return err;
418  }
419 
420  // Check number of channels
421  if (nof_channels_ % nof_channels != 0 or nof_channels_ / nof_channels > nof_radios) {
422  Error("Number of requested channels (" + std::to_string(nof_channels_) +
423  ") is different than the RFNOC "
424  "available channels (" +
425  std::to_string(nof_radios * nof_channels) + ")");
426  return UHD_ERROR_VALUE;
427  }
428 
429  // Re-calculate number of radios
430  nof_radios = nof_channels_ / nof_channels;
431 
432  // Make USRP
433  err = usrp_multi_make(device_addr);
434  if (err != UHD_ERROR_NONE) {
435  return err;
436  }
437 
438  // Reset blocks after a stream
439  device3->clear();
440 
441  // Create control interfaces
442  err = create_control_interfaces();
443  if (err != UHD_ERROR_NONE) {
444  return err;
445  }
446 
447  // Connect components
448  if (loopback) {
449  err = connect_loopback();
450  } else {
451  err = connect();
452  }
453  if (err != UHD_ERROR_NONE) {
454  return err;
455  }
456 
457  return UHD_ERROR_NONE;
458  }
