167 auto find_rrc_obj_id_gap(const Container& c) -> decltype(get_asn1_obj_id(c[0]))
168 {
169  auto id_cmp_op = rrc_obj_id_cmp{};
170  assert(std::is_sorted(c.begin(), c.end(), id_cmp_op));
171 
172  auto prev_it = c.begin();
173  if (prev_it != c.end() and get_asn1_obj_id(*prev_it) == 1) {
174  auto it = prev_it;
175  for (++it; it != c.end(); prev_it = it, ++it) {
176  if (get_asn1_obj_id(*it) > get_asn1_obj_id(*prev_it) + 1) {
177  break;
178  }
179  }
180  }
181  return (prev_it == c.end()) ? 1 : get_asn1_obj_id(*prev_it) + 1; // starts at 1.

282 template <typename List, typename RemFunctor, typename AddFunctor, typename ModFunctor>
283 void compute_cfg_diff(const List& src_list,
284  const List& target_list,
285  RemFunctor rem_func,
286  AddFunctor add_func,
287  ModFunctor mod_func)
288 {
289  auto id_cmp_op = rrc_obj_id_cmp{};
290  assert(std::is_sorted(src_list.begin(), src_list.end(), id_cmp_op));
291  assert(std::is_sorted(target_list.begin(), target_list.end(), id_cmp_op));
292 
293  auto src_it = src_list.begin();
294  auto target_it = target_list.begin();
295 
296  bool src_left = src_it != src_list.end();
297  bool target_left = target_it != target_list.end();
298  while (src_left or target_left) {
299  if (not target_left or (src_left and id_cmp_op(*src_it, *target_it))) {
300  rem_func(src_it++);
301  } else if (not src_left or (target_left and id_cmp_op(*target_it, *src_it))) {
302  // a new object has been added to target
303  add_func(target_it++);
304  } else {
305  mod_func(src_it++, target_it++);
306  }
307  src_left = src_it != src_list.end();
308  target_left = target_it != target_list.end();
309  }
310 }
