151 bool paging_manager::add_paging_record(uint32_t ueid, const asn1::rrc::paging_record_s& paging_record)
152 {
153  constexpr static const int sf_pattern[4][4] = {{9, 4, -1, 0}, {-1, 9, -1, 4}, {-1, -1, -1, 5}, {-1, -1, -1, 9}};
154 
155  ueid = ((uint32_t)ueid) % 1024;
156  uint32_t i_s = (ueid / N) % Ns;
157 
158  int sf_idx = sf_pattern[i_s % 4][(Ns - 1) % 4];
159  if (sf_idx < 0) {
160  logger.error("SF pattern is N/A for Ns=%d, i_s=%d, imsi_decimal=%d", Ns, i_s, ueid);
161  return false;
162  }
163  size_t sf_key = static_cast<size_t>(get_sf_idx_key(sf_idx));
164 
165  subframe_info& locked_sf = sf_pending_pcch[sf_key];
166  std::lock_guard<std::mutex> lock(locked_sf.mutex);
167 
168  size_t sfn_cycle_idx = ((size_t)T / (size_t)N) * (size_t)(ueid % N);
169  pcch_info& pending_pcch = locked_sf.pending_paging[sfn_cycle_idx];
170  auto& record_list = pending_pcch.pcch_msg.msg.c1().paging().paging_record_list;
171 
172  if (record_list.size() >= ASN1_RRC_MAX_PAGE_REC) {
173  logger.warning("Failed to add new paging record for ueid=%d. Cause: no paging record space left.", ueid);
174  return false;
175  }
176  if (pending_pcch.is_tx()) {
177  logger.error("Adding Paging records to ueid=%d PCCH that has been already sent but not cleared.", ueid);
178  pending_pcch.clear();
179  }
180 
181  if (pending_pcch.pdu == nullptr) {
182  pending_pcch.pdu = srsran::make_byte_buffer();
183  if (pending_pcch.pdu == nullptr) {
184  logger.warning("Failed to add new paging record for ueid=%d. Cause: No buffers available", ueid);
185  return false;
186  }
187  }
188 
189  record_list.push_back(paging_record);
190 
191  asn1::bit_ref bref(pending_pcch.pdu->msg, pending_pcch.pdu->get_tailroom());
192  if (pending_pcch.pcch_msg.pack(bref) == asn1::SRSASN_ERROR_ENCODE_FAIL) {
193  logger.error("Failed to pack PCCH message");
194  pending_pcch.clear();
195  return false;
196  }
197  pending_pcch.pdu->N_bytes = (uint32_t)bref.distance_bytes();
198 
199  return true;
200 }

202 size_t paging_manager::pending_pcch_bytes(tti_point tti_tx_dl)
203 {
204  int sf_key = get_sf_idx_key(tti_tx_dl.sf_idx());
205  if (sf_key < 0) {
206  // tti_tx_dl is not in a paging subframe
207  return 0;
208  }
209 
210  subframe_info& locked_sf = sf_pending_pcch[static_cast<size_t>(sf_key)];
211  std::unique_lock<std::mutex> lock(locked_sf.mutex, std::try_to_lock);
212  if (not lock.owns_lock()) {
213  // If the scheduler fails to lock, it will postpone the PCCH transmission to the next paging cycle
214  return 0;
215  }
216 
217  // clear old PCCH that has been transmitted at this point
218  while (not locked_sf.transmitted_pcch.empty() and locked_sf.transmitted_pcch.front()->tti_tx_dl < tti_tx_dl) {
219  locked_sf.transmitted_pcch.front()->clear();
220  locked_sf.transmitted_pcch.pop_front();
221  }
222 
223  const pcch_info& pending_pcch = locked_sf.pending_paging[tti_tx_dl.sfn() % T];
224  if (pending_pcch.empty()) {
225  return 0;
226  }
227  return pending_pcch.pdu->size();
228 }

230 template <typename Callable>
231 bool paging_manager::read_pdu_pcch(tti_point tti_tx_dl, const Callable& func)
232 {
233  int sf_key = get_sf_idx_key(tti_tx_dl.sf_idx());
234  if (sf_key < 0) {
235  logger.warning("%s was called for tti=%d, which is not a paging subframe index", __FUNCTION__, tti_tx_dl.to_uint());
236  return false;
237  }
238 
239  subframe_info& locked_sf = sf_pending_pcch[static_cast<size_t>(sf_key)];
240  std::lock_guard<std::mutex> lock(locked_sf.mutex);
241 
242  pcch_info& pending_pcch = locked_sf.pending_paging[tti_tx_dl.sfn() % T];
243 
244  if (pending_pcch.empty()) {
245  logger.warning("read_pdu_pdcch(...) called for tti=%d, but there is no pending pcch message", tti_tx_dl.to_uint());
246  return false;
247  }
248 
249  // Call callable for existing PCCH pdu
250  if (func(*pending_pcch.pdu, pending_pcch.pcch_msg, not pending_pcch.is_tx())) {
251  if (not pending_pcch.is_tx()) {
252  // first tx. Enqueue in list of transmitted pcch. We do not erase the PCCH yet because it may be transmitted
253  // by other carriers
254  pending_pcch.tti_tx_dl = tti_tx_dl;
255  locked_sf.transmitted_pcch.push_back(&pending_pcch);
256  }
257  return true;
258  }
259  return false;
260 }
