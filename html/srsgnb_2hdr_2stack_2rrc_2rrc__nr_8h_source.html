182  srsran::unique_byte_buffer_t pack_into_pdu(const T& msg, const char* context_name = nullptr)
183  {
184  context_name = context_name == nullptr ? __FUNCTION__ : context_name;
185  // Allocate a new PDU buffer and pack the
186  srsran::unique_byte_buffer_t pdu = srsran::make_byte_buffer();
187  if (pdu == nullptr) {
188  logger.error("Couldn't allocate PDU in %s.", context_name);
189  return nullptr;
190  }
191  asn1::bit_ref bref(pdu->msg, pdu->get_tailroom());
192  if (msg.pack(bref) == asn1::SRSASN_ERROR_ENCODE_FAIL) {
193  logger.error("Failed to pack message in %s. Discarding it.", context_name);
194  return nullptr;
195  }
196  pdu->N_bytes = bref.distance_bytes();
