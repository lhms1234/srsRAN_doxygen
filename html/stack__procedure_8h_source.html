80  callback_id_t on_next_call(callback_t f_)
81  {

83  func_list[idx].func = std::move(f_);
84  func_list[idx].call_always = false;
85  return idx;
86  }
87  callback_id_t on_every_call(callback_t f_)
88  {
89  uint32_t idx = get_new_callback();
90  func_list[idx].func = std::move(f_);
91  func_list[idx].call_always = true;
92  return idx;
93  }
94 
95  // call all callbacks
96  template <typename... ArgsRef>
97  void operator()(ArgsRef&&... args)
98  {
99  for (auto& f : func_list) {
100  if (f.active) {
101  f.func(std::forward<ArgsRef>(args)...);
102  if (not f.call_always) {
103  f.active = false;
104  }
105  }
106  }
107  }
108 
109 private:
110  uint32_t get_new_callback()
111  {
112  uint32_t i = 0;
113  for (; i < func_list.size() and func_list[i].active; ++i) {
114  }
115  if (i == func_list.size()) {
116  func_list.emplace_back();
117  }
118  func_list[i].active = true;
119  return i;
120  }

200 /**************************************************************************************
201  * class: proc_base_t
202  * Provides a polymorphic interface for resumable procedures. This base can then be used
203  * by a "proc_manager_list_t" via the virtual method "proc_base_t::run()".
204  * With public methods:
205  * - run() - executes proc_t<T>::step(), and updates procedure state.
206  * - is_busy()/is_idle() - tells if procedure is currently running. Busy procedures
207  * cannot be re-launched
208  * - then() - called automatically when a procedure has finished. Useful for actions
209  * upon procedure completion, like sending back a response or logging.
210  * With protected methods:
211  * - step() - method overriden by child class that will be called by run(). step()
212  * executes a procedure "action" based on its current internal state,
213  * and return a proc_outcome_t variable with possible values:
214  * - yield - the procedure performed the action but hasn't completed yet.
215  * - error - the procedure has finished unsuccessfully
216  * - success - the procedure has completed successfully
217  ************************************************************************************/
218 class proc_base_t
219 {
220 public:
221  virtual ~proc_base_t() = default;
222 
224  bool run()
225  {
226  if (is_busy()) {
227  proc_outcome_t outcome = step();
228  handle_outcome(outcome);
229  }
230  return is_busy();
231  }
232 
234  bool is_busy() const { return proc_state == proc_status_t::on_going; }
235  bool is_idle() const { return proc_state == proc_status_t::idle; }
236 
237 protected:
238  enum class proc_status_t { idle, on_going };
239  virtual proc_outcome_t step() = 0;
240  virtual void run_then(bool is_success) = 0;

253 /**************************************************************************************
254  * class: proc_t<T, ResultType>
255  * Manages the lifetime of a procedure of type T, including its alloc, launching,
256  * and reset back to "inactive" state once the procedure has been completed.
257  * The result of a procedure run is of type "proc_result_t<ResultType>". ResultType has
258  * to coincide with the type returned by the method "T::get_result()".
259  * There are three main ways to use the result of a procedure run:
260  * - "T::then(const proc_result_t<T>&)" - method in T that runs on completion, and
261  * gets as argument the result of the run
262  * - "proc_t<T>::get_future()" - returns a proc_future_t<T> which the user can use
263  * directly to check the result of a run
264  * - "proc_t<T>::then/then_always()" - provide dynamically a continuation task, for
265  * instance, by providing a lambda
266  * It uses a unique_ptr<T> to allow the use of procedures that are forward declared.
267  ************************************************************************************/

432  * - add_proc(...) - adds a proc_t<T>, and once the procedure has completed, takes it

436  * specifies a proc_impl_t step() function
