95  void* allocate_node(size_t sz)
96  {
97  srsran_assert(sz <= ObjSize, "Allocated node size=%zd exceeds max object size=%zd", sz, ObjSize);
98  worker_ctxt* worker_ctxt = get_worker_cache();
99 
100  void* node = worker_ctxt->cache.try_pop();
101  if (node == nullptr) {
102  // fill the thread local cache enough for this and next allocations
103  std::array<void*, batch_steal_size> popped_blocks;
104  size_t n = central_mem_cache.try_pop(popped_blocks);
105  for (size_t i = 0; i < n; ++i) {
106  new (popped_blocks[i]) obj_storage_t();
107  worker_ctxt->cache.push(static_cast<void*>(popped_blocks[i]));
108  }
109  node = worker_ctxt->cache.try_pop();
110  }

137

139  // if local cache reached max capacity, send half of the blocks to central cache
