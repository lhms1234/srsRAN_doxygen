174 template <typename... Args>
175 void log_rlc_am_nr_status_pdu_to_string(srslog::log_channel& log_ch,
176  const char* fmt_str,
177  rlc_am_nr_status_pdu_t* status,
178  const std::string& rb_name,
179  Args&&... args)
180 {
181  if (not log_ch.enabled()) {
182  return;
183  }
184  fmt::memory_buffer buffer;
185  fmt::format_to(buffer, "ACK_SN = {}, N_nack = {}", status->ack_sn, status->nacks.size());
186  if (status->nacks.size() > 0) {
187  fmt::format_to(buffer, ", NACK_SN = ");
188  for (uint32_t i = 0; i < status->nacks.size(); ++i) {
189  if (status->nacks[i].has_nack_range) {
190  if (status->nacks[i].has_so) {
191  fmt::format_to(buffer,
192  "[{} {}:{} r{}]",
193  status->nacks[i].nack_sn,
194  status->nacks[i].so_start,
195  status->nacks[i].so_end,
196  status->nacks[i].nack_range);
197  } else {
198  fmt::format_to(buffer, "[{} r{}]", status->nacks[i].nack_sn, status->nacks[i].nack_range);
199  }
200  } else {
201  if (status->nacks[i].has_so) {
202  fmt::format_to(
203  buffer, "[{} {}:{}]", status->nacks[i].nack_sn, status->nacks[i].so_start, status->nacks[i].so_end);
204  } else {
205  fmt::format_to(buffer, "[{}]", status->nacks[i].nack_sn);
206  }
207  }
208  }
209  }
210  log_ch(fmt_str, std::forward<Args>(args)..., to_c_str(buffer));
211 }
