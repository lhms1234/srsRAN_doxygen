38 class intra_measure_base : public srsran::thread
39 {
40  /*
41  * The intra-cell measurement has 5 different states:
42  * - idle: it has been initiated and it is waiting to get configured to start capturing samples. From any state
43  * except quit can transition to idle.
44  * - wait: waits for the TTI trigger to transition to receive
45  * - receive: captures base-band samples for intra_freq_meas_len_ms and goes to measure.
46  * - measure: enables the inner thread to start the measuring function. The asynchronous buffer will transition to
47  * wait as soon as it has read the data from the buffer.
48  * - quit: stops the inner thread and quits. Transition from any state measure state.
49  *
50  * FSM abstraction:
51  *
52  * +------+ set_cells_to_meas +------+ receive_tti_trigger +---------+
53  * | Idle | --------------------->| Wait |------------------------------>| Receive |
54  * +------+ +------+ +---------+
55  * ^ ^ | stop +------+
56  * | Read buffer | | ----->| Quit |
57  * init +---------+ intra_freq_meas_len_ms | +------+
58  * meas_stop | Measure |<----------------------------------+
59  * +---------+
60  *
61  * This class has been designed to be thread safe. Any method can be called from different threads as long as
62  * init_generic is called when the FSM is in idle.
63  */

242  bool receive_tti_trigger(uint32_t tti)
243  {
244  std::lock_guard<std::mutex> lock(mutex);
245 
246  // If the elapsed time does not satisfy with the minimum time, do not trigger
247  uint32_t elapsed_tti = TTI_SUB(tti, last_measure_tti);
248  if (elapsed_tti < context.meas_period_ms and state.get_state() != internal_state::wait_first) {
249  return false;
250  }
251 
252  // If the TTI period is not configured, it will be always true
253  if (context.trigger_tti_period == 0) {
254  return true;
255  }
256 
257  // Check if trigger condition is satisfied
258  return tti % context.trigger_tti_period == context.trigger_tti_offset;
259  }
