52 struct rlc_am_nr_tx_state_t {
53  /*
54  * TX_Next_Ack: This state variable holds the value of the SN of the next RLC SDU for which a positive
55  * acknowledgment is to be received in-sequence, and it serves as the lower edge of the transmitting window. It is
56  * initially set to 0, and is updated whenever the AM RLC entity receives a positive acknowledgment for an RLC SDU
57  * with SN = TX_Next_Ack.
58  */
59  uint32_t tx_next_ack;
60  /*
61  * TX_Next: This state variable holds the value of the SN to be assigned for the next newly generated AMD PDU. It is
62  * initially set to 0, and is updated whenever the AM RLC entity constructs an AMD PDU with SN = TX_Next and
63  * contains an RLC SDU or the last segment of a RLC SDU.
64  */
65  uint32_t tx_next;
66  /*
67  * POLL_SN: This state variable holds the value of the highest SN of the AMD PDU among the AMD PDUs submitted to
68  * lower layer when POLL_SN is set according to sub clause 5.3.3.2. It is initially set to 0.
69  */
70  uint32_t poll_sn;
71  /*
72  * PDU_WITHOUT_POLL: This counter is initially set to 0. It counts the number of AMD PDUs sent since the most recent
73  * poll bit was transmitted.
74  */
75  uint32_t pdu_without_poll;
76  /*
77  * BYTE_WITHOUT_POLL: This counter is initially set to 0. It counts the number of data bytes sent since the most
78  * recent poll bit was transmitted.
79  */
80  uint32_t byte_without_poll;
81 };
