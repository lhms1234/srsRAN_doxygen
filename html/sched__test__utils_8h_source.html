278  tti_ev::user_cfg_ev* get_user_cfg(uint16_t rnti)
279  {
280  std::vector<tti_ev::user_cfg_ev>& user_updates = tti_events[tti_counter].user_updates;
281  auto it = std::find_if(
282  user_updates.begin(), user_updates.end(), [&rnti](tti_ev::user_cfg_ev& user) { return user.rnti == rnti; });
283  if (it == user_updates.end()) {

286  return &user_updates.back();

293  void rem_old_users()
294  {
295  // remove users that pass their connection duration
296  for (auto it = current_users.begin(); it != current_users.end();) {
297  user_data& user = it->second;
298  uint32_t rem_tti = user.tti_start + user.tti_duration;
299  if (rem_tti > tti_counter) {
300  ++it;
301  continue;
302  }
303  // set the call rem_user(...) at the right tti
304  auto& l = tti_events[rem_tti].user_updates;
305  auto user_it = std::find_if(l.begin(), l.end(), [&it](tti_ev::user_cfg_ev& u) { return it->first == u.rnti; });
306  if (user_it == l.end()) {
307  l.emplace_back();
308  l.back().rem_user = true;
309  l.back().rnti = it->first;
310  } else {
311  user_it->rem_user = true;
312  }
313  it = current_users.erase(it);
314  }
315  }
