52  int handle_message(const unique_byte_array_t& rx_buf, const uint32_t n)
53  {
54  logger.debug("Received %d B from remote.", n);
55 
56  // Chop incoming msg, first two bytes are length of the JSON
57  // (see IPL4_EUTRA_SYSTEM_Definitions.ttcn
58  uint16_t json_len = ((uint16_t)rx_buf->at(0) << 8) | rx_buf->at(1);
59 
60  // Copy JSON from received buffer and null-terminate
61  char json[json_len + 1];
62  memcpy(json, &rx_buf->at(2), json_len);
63  json[json_len] = '\0';
64 
65  // The data part after the JSON starts right here but handling
66  // is done in the respective functions
67  uint16_t rx_buf_offset = json_len + 2;
68 
69  Document document;
70  if (document.Parse(json).HasParseError() || document.IsObject() == false) {
71  logger.error((uint8_t*)json, json_len, "Error parsing incoming data.");
72  return SRSRAN_ERROR;
73  }
74 
75  // Pretty-print
76  StringBuffer buffer;
77  PrettyWriter<StringBuffer> writer(buffer);
78  document.Accept(writer);
79  logger.info("Received %d bytes\n%s", json_len, (char*)buffer.GetString());
80 
81  // check for common
82  assert(document.HasMember("Common"));
83  assert(document["Common"].IsObject());
84 
85  // Check for request type
86  assert(document.HasMember("Request"));
87  assert(document["Request"].IsObject());
88 
89  // Get request type
90  const Value& request = document["Request"];
91  if (request.HasMember("Cell")) {
92  logger.info("Received Cell request.");
93  handle_request_cell(document, &rx_buf->at(rx_buf_offset), n - rx_buf_offset);
94  } else if (request.HasMember("L1MacIndCtrl")) {
95  logger.info("Received L1MacIndCtrl request.");
96  handle_request_l1_mac_ind_ctrl(document);
97  } else if (request.HasMember("RadioBearerList")) {
98  logger.info("Received RadioBearerList request.");
99  handle_request_radio_bearer_list(document);
100  } else if (request.HasMember("CellAttenuationList")) {
101  logger.info("Received CellAttenuationList request.");
102  handle_request_cell_attenuation_list(document);
103  } else if (request.HasMember("PdcpCount")) {
104  logger.info("Received PdcpCount request.");
105  handle_request_pdcp_count(document);
106  } else if (request.HasMember("AS_Security")) {
107  logger.info("Received AS_Security request.");
108  handle_request_as_security(document);
109  } else if (request.HasMember("EnquireTiming")) {
110  logger.info("Received EnquireTiming request.");
111  handle_request_enquire_timing(document);
112  } else if (request.HasMember("Paging")) {
113  logger.info("Received Paging request.");
114  handle_request_paging(document, &rx_buf->at(rx_buf_offset), n - rx_buf_offset);
115  } else if (request.HasMember("PdcpHandoverControl")) {
116  logger.info("Received PdcpHandoverControl.");
117  handle_request_pdcp_handover_control(document);
118  } else {
119  logger.error("Received unknown request.");
120  }
121 
122  return SRSRAN_SUCCESS;
