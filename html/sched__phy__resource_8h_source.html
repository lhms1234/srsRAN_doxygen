130 inline uint32_t count_prb_per_tb(const rbgmask_t& bitmask)
131 {
132  uint32_t Nprb = cell_nof_rbg_to_prb(bitmask.size());
133  uint32_t P = srsran_ra_type0_P(Nprb);
134  uint32_t nof_prb = P * bitmask.count();
135  if (bitmask.test(bitmask.size() - 1)) {
136  nof_prb -= bitmask.size() * P - Nprb;
137  }
138  return nof_prb;
