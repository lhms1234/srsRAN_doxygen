398 bool nowhitespace_string_to_enum(EnumType& e, const std::string& s)
399 {
400  std::string s_nows = s;
401  std::transform(s_nows.begin(), s_nows.end(), s_nows.begin(), ::tolower);
402  s_nows.erase(std::remove(s_nows.begin(), s_nows.end(), ' '), s_nows.end());
403  s_nows.erase(std::remove(s_nows.begin(), s_nows.end(), '-'), s_nows.end());
404  for (uint32_t i = 0; i < EnumType::nof_types; ++i) {
405  e = (typename EnumType::options)i;
406  std::string s_nows2 = e.to_string();
407  std::transform(s_nows2.begin(), s_nows2.end(), s_nows2.begin(), ::tolower);
408  s_nows2.erase(std::remove(s_nows2.begin(), s_nows2.end(), ' '), s_nows2.end());
409  s_nows2.erase(std::remove(s_nows2.begin(), s_nows2.end(), '-'), s_nows2.end());
410  if (s_nows2 == s_nows) {
411  return true;
412  }
413  }
414  return false;

439 template <typename EnumType>
440 int number_to_enum(EnumType& enum_val, Setting& root)
441 {
442  if (root.isNumber()) {
443  typename EnumType::number_type val = 0;
444  if (root.getType() == Setting::TypeInt64) {
445  val = (long int)root;
446  } else if (root.getType() == Setting::TypeInt) {
447  val = (int)root;
448  }
449  bool found = asn1::number_to_enum(enum_val, val);
450  if (not found) {
451  std::ostringstream ss;
452  ss << val;
453  fprintf(stderr, "Invalid option: %s for enum field \"%s\"\n", ss.str().c_str(), root.getName());
454  ss.str("");
455  ss << EnumType((typename EnumType::options)0).to_number();
456  fprintf(stderr, "Valid options: %s", ss.str().c_str());
457  for (uint32_t i = 1; i < EnumType::nof_types; i++) {
458  ss.str("");
459  ss << EnumType((typename EnumType::options)i).to_number();
460  fprintf(stderr, ", %s", ss.str().c_str());
461  }
462  fprintf(stderr, "\n");
463  }
464  return found ? 0 : -1;
465  } else {
466  std::string str_val = root.c_str();

468  }
