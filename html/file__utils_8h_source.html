43 inline std::pair<std::string, std::string> split_filename_extension(const std::string& filename)
44 {
45  // Search for the last dot.
46  auto dot_pos = filename.find_last_of('.');
47 
48  // Check for the following corner cases (returns {filename, ""}):
49  // a) No dot found: my_file
50  // b) Dot found at the beginning: .my_file
51  // c) Dot found at the end: my_file.
52  if (dot_pos == std::string::npos || dot_pos == 0 || dot_pos == filename.size() - 1) {
53  return {filename, ""};
54  }
55 
56  // Handle directories that contain dots, search for the last separator
57  // character.
58  auto separator_pos = filename.find_last_of('/');
59 
60  // Check for the following corner cases (returns {filename, ""}):
61  // a) /my_folder.1/my_file
62  // b) /my_folder.1/.my_file
63  if (separator_pos != std::string::npos && separator_pos >= dot_pos - 1) {
64  return {filename, ""};
65  }
66 
67  return {filename.substr(0, dot_pos), filename.substr(dot_pos)};
68 }
