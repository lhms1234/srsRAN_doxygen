107  void run_tti(uint32_t tti)
108  {
109  std::unique_lock<std::mutex> lock(mutex);
110 
111  switch (activation_state) {
112  case idle:
113  // waiting for receiving a command, do nothing
114  break;
115  case waiting:
116  // Detect that TTI when the CMD needs to be applied, the activation cannot be done instantly because some
117  // workers might be currently ongoing, so only update state
118  if (_tti_greater_or_equal_than(tti, activation_tti)) {
119  activation_state = transition;
120  }
121  break;
122  case transition:
123  // Detect when the TTI has increased enough to make sure there arent workers, set the configuration
124  if (TTI_SUB(tti, activation_tti) >= activation_margin_tti) {
125  // Reload cell states
126  for (uint32_t i = 1; i < SRSRAN_MAX_CARRIERS; i++) {
127  // Get Activation command value
128  bool activate = _get_cmd_activation(i);
129 
130  // Apply activation only if the cell was configured
131  if (scell_cfg[i].status != cfg::none) {
132  scell_cfg[i].status = activate ? cfg::active : cfg::inactive;
133  }
134  }
135 
136  // Go back to initial state
137  activation_state = idle;
138  }
139  break;
140  }
