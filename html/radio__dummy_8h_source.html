114  void advance_tx_timestamp(uint64_t ts, bool round_sf = false)
115  {
116  std::lock_guard<std::mutex> lock(tx_mutex);
117 
118  // Make sure new timestamp has not passed
119  if (ts < tx_timestamp) {
120  return;
121  }
122 
123  // Calculate transmission gap
124  uint32_t tx_gap = (uint32_t)(ts - tx_timestamp);
125 
126  // Round gap to subframe size
127  if (round_sf) {
128  uint64_t sf_sz = (uint64_t)(srate_hz / 1e3);
129  tx_gap = sf_sz * SRSRAN_CEIL(tx_gap, sf_sz);
130  }
131 
132  // Skip zeros if there is no gap
133  if (tx_gap == 0) {
134  return;
135  }
136 
137  // Write zeros in tx ring buffer
138  write_zeros_ring_buffers(tx_ring_buffers, tx_gap);
139 
140  // Update new transmit timestamp
141  tx_timestamp += tx_gap;
142  }
