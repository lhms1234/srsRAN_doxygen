117  int get_grant_avg_cqi(rbg_interval interv) const
118  {
119  if (not subband_cqi_enabled()) {
120  return static_cast<int>(wb_cqi_avg);
121  }
122  float cqi = 0;
123  uint32_t sbstart = rbg_to_sb_index(interv.start()), sbend = rbg_to_sb_index(interv.stop() - 1) + 1;
124  for (uint32_t sb = sbstart; sb < sbend; ++sb) {

137  int get_grant_avg_cqi(const rbgmask_t& mask) const
138  {
139  if (not subband_cqi_enabled()) {
140  return static_cast<int>(wb_cqi_avg);
141  }
142  float cqi = 0;
143  uint32_t count = 0;
144  for (int rbg = mask.find_lowest(0, mask.size()); rbg != -1; rbg = mask.find_lowest(rbg, mask.size())) {
145  uint32_t sb = rbg_to_sb_index(rbg);
146  cqi += get_subband_cqi_(sb);
147  count++;
148  rbg = static_cast<int>(((sb + 1U) * cell_nof_rbg + N() - 1U) / N()); // skip to next subband index
149  }
150  return static_cast<int>(cqi / count);
151  }
