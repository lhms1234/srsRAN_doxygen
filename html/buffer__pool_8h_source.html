49 template <class buffer_t>
50 class buffer_pool
51 {
52 public:
53  // non-static methods
54  buffer_pool(int capacity_ = -1)
55  {
56  uint32_t nof_buffers = POOL_SIZE;
57  if (capacity_ > 0) {
58  nof_buffers = (uint32_t)capacity_;
59  }
60  pool.reserve(nof_buffers);
61  free_list.reserve(nof_buffers);
62  pthread_mutex_init(&mutex, nullptr);
63  pthread_cond_init(&cv_not_empty, nullptr);
64  for (uint32_t i = 0; i < nof_buffers; i++) {
65  buffer_t* b = new (std::nothrow) buffer_t;
66  if (!b) {
67  perror("Error allocating memory. Exiting...\n");
68  exit(-1);
69  }
70  pool.push_back(b);
71  free_list.push_back(b);
72  }
73  capacity = nof_buffers;
74  }
75 
76  ~buffer_pool()
77  {
78  for (auto* p : pool) {
79  delete p;
80  }
81  pthread_cond_destroy(&cv_not_empty);
82  pthread_mutex_destroy(&mutex);
83  }

106  buffer_t* allocate(const char* debug_name = nullptr, bool blocking = false)
107  {
108  pthread_mutex_lock(&mutex);
109  buffer_t* b = nullptr;
110 
111  if (!free_list.empty()) {
112  b = free_list.back();
113  free_list.pop_back();
114 
115  if (is_almost_empty()) {
116  printf("Warning buffer pool capacity is %f %%\n", (float)100 * free_list.size() / capacity);
117  }
118 #ifdef SRSRAN_BUFFER_POOL_LOG_ENABLED
119  if (debug_name) {
120  strncpy(b->debug_name, debug_name, SRSRAN_BUFFER_POOL_LOG_NAME_LEN);
121  b->debug_name[SRSRAN_BUFFER_POOL_LOG_NAME_LEN - 1] = 0;
122  }
123 #endif
124  } else if (blocking) {
125  // blocking allocation
126  while (free_list.empty()) {
127  pthread_cond_wait(&cv_not_empty, &mutex);
128  }
129 
130  // retrieve the new buffer
131  b = free_list.back();
132  free_list.pop_back();
133 
134  // do not print any warning
135  } else {
136  printf("Error - buffer pool is empty\n");
137 
138 #ifdef SRSRAN_BUFFER_POOL_LOG_ENABLED
139  print_all_buffers();
140 #endif
141  }
142 
143  pthread_mutex_unlock(&mutex);
144  return b;
145  }

147  bool deallocate(buffer_t* b)
148  {
149  bool ret = false;
150  pthread_mutex_lock(&mutex);
151  if (std::find(pool.cbegin(), pool.cend(), b) != pool.cend()) {
152  free_list.push_back(b);
153  ret = true;
154  }
155  pthread_cond_signal(&cv_not_empty);
156  pthread_mutex_unlock(&mutex);
