113  void run_tti(const uint32_t tti, const uint32_t tti_jump) override
114  {
115  // Wait for tick from test bench
116  std::unique_lock<std::mutex> lock(pending_tti_mutex);
117  while (not pending_tti and running) {
118  pending_tti_cvar.wait_for(lock, std::chrono::milliseconds(1));
119  }
120 
121  // Let the tick proceed
122  pending_tti = false;
123  pending_tti_cvar.notify_all();
124 
125  // Run PRACH
126  if (prach_period != 0) {
127  uint32_t slot_idx = tti % SRSRAN_NSLOTS_PER_FRAME_NR(srsran_subcarrier_spacing_15kHz);
128  uint32_t sfn = tti / SRSRAN_NSLOTS_PER_FRAME_NR(srsran_subcarrier_spacing_15kHz);
129  if (not prach_pending and slot_idx == 0 and sfn % prach_period == 0) {
130  prach_preamble = srsran_random_uniform_int_dist(random_gen, 0, 63);
131  phy.send_prach(0, prach_preamble, 0.0f, 0.0f);
132  prach_pending = true;
133  }
134  }
135  }
