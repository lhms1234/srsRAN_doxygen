313  // Enqueues the Rx PDU in the resume queue
314  void queue_rx_pdu(uint8_t* payload, uint32_t nof_bytes)
315  {
316  unique_byte_buffer_t rx_pdu = srsran::make_byte_buffer();
317  if (rx_pdu == nullptr) {
318  srslog::fetch_basic_logger("RLC").warning("Couldn't allocate PDU in %s().", __FUNCTION__);
319  return;
320  }
321 
322  if (rx_pdu->get_tailroom() < nof_bytes) {
323  srslog::fetch_basic_logger("RLC").warning("Not enough space to store PDU.");
324  return;
325  }
326 
327  memcpy(rx_pdu->msg, payload, nof_bytes);
328  rx_pdu->N_bytes = nof_bytes;
329 
330  // Do not block ever
331  if (!rx_pdu_resume_queue.try_push(std::move(rx_pdu))) {
332  srslog::fetch_basic_logger("RLC").warning("Dropping SDUs while bearer suspended.");
333  return;
334  }
335  }
336 
337  // Enqueues the Tx SDU in the resume queue
338  void queue_tx_sdu(unique_byte_buffer_t sdu)
339  {
340  // Do not block ever
341  if (not tx_sdu_resume_queue.try_push(std::move(sdu))) {
342  srslog::fetch_basic_logger("RLC").warning("Dropping SDUs while bearer suspended.");
343  return;
344  }
345  }
346 
347  static_blocking_queue<unique_byte_buffer_t, 256> rx_pdu_resume_queue;
348  static_blocking_queue<unique_byte_buffer_t, 256> tx_sdu_resume_queue;
