229  // Section 6.1.2
230  uint32_t parse_packet(uint8_t* ptr)
231  {
232  uint8_t* init_ptr = ptr;
233  nof_subheaders = 0;
234  bool ret = false;
235  do {
236  if (nof_subheaders < (int)max_subheaders) {
237  ret = subheaders[nof_subheaders].read_subheader(&ptr);
238  nof_subheaders++;
239  }
240 
241  if (ret && ((ptr - init_ptr) >= (int32_t)pdu_len)) {
242  // stop processing last subheader indicates another one but all bytes are consumed
243  nof_subheaders = 0;
244  logger.warning(init_ptr, pdu_len, "Corrupted MAC PDU - all bytes have been consumed (pdu_len=%d)", pdu_len);
245  return SRSRAN_ERROR;
246  }
247  } while (ret && (nof_subheaders + 1) < (int)max_subheaders && ((int32_t)pdu_len > (ptr - init_ptr)));
248 
249  for (int i = 0; i < nof_subheaders; i++) {
250  subheaders[i].read_payload(&ptr);
251 
252  // stop processing if we read payload beyond the PDU length
253  if ((ptr - init_ptr) > (int32_t)pdu_len) {
254  nof_subheaders = 0;
255  logger.warning(init_ptr, pdu_len, "Corrupted MAC PDU - all bytes have been consumed (pdu_len=%d)", pdu_len);

257  }
258  }
259  return SRSRAN_SUCCESS;
