66 struct zerobit_counter {
67  static Integer msb_count(Integer value)
68  {
69  if (value == 0) {
70  return std::numeric_limits<Integer>::digits;
71  }
72  Integer ret = 0;
73  for (Integer shift = std::numeric_limits<Integer>::digits >> 1; shift != 0; shift >>= 1) {
74  Integer tmp = value >> shift;
75  if (tmp != 0) {
76  value = tmp;
77  } else {
78  ret |= shift;
79  }
80  }
81  return ret;
82  }
83  static Integer lsb_count(Integer value)
84  {
85  if (value == 0) {
86  return std::numeric_limits<Integer>::digits;
87  }
88  Integer ret = 0;
89  for (Integer shift = std::numeric_limits<Integer>::digits >> 1, mask = std::numeric_limits<Integer>::max() >> shift;
90  shift != 0;
91  shift >>= 1, mask >>= shift) {
92  if ((value & mask) == 0) {
93  value >>= shift;
94  ret |= shift;
95  }
96  }
97  return ret;
98  }
