65  int handle_message(const unique_byte_array_t& rx_buf, const uint32_t n)
66  {
67  logger.debug(rx_buf->begin(), n, "Received %d B from remote.", n);
68 
69  // Chop incoming msg, first two bytes are length of the JSON
70  // (see IPL4_EUTRA_SYSTEM_Definitions.ttcn
71  uint16_t json_len = ((uint16_t)rx_buf->at(0) << 8) | rx_buf->at(1);
72 
73  // The data part after the JSON starts right here but handling
74  // is done in the respective functions
75  uint16_t rx_buf_offset = json_len + 2;
76 
77  Document document;
78  if (document.Parse((char*)&rx_buf->at(2)).HasParseError() || document.IsObject() == false) {
79  logger.error((uint8_t*)&rx_buf->at(2), json_len, "Error parsing incoming data.");
80  return SRSRAN_ERROR;
81  }
82 
83  // Pretty-print
84  StringBuffer buffer;
85  PrettyWriter<StringBuffer> writer(buffer);
86  document.Accept(writer);
87  logger.info("Received JSON with %d B\n%s", json_len, (char*)buffer.GetString());
88 
89  // check for common
90  assert(document.HasMember("Common"));
91  assert(document["Common"].IsObject());
92 
93  // Check for request type
94  assert(document.HasMember("RrcPdu"));
95  assert(document["RrcPdu"].IsObject());
96 
97  // Get request type
98  const Value& rrcpdu = document["RrcPdu"];
99  if (rrcpdu.HasMember("Ccch")) {
100  rx_buf_offset += 2;
101  handle_ccch_pdu(document, &rx_buf->at(rx_buf_offset), n - rx_buf_offset);
102  } else if (rrcpdu.HasMember("Dcch")) {
103  rx_buf_offset += 2;
104  uint32_t lcid = document["Common"]["RoutingInfo"]["RadioBearerId"]["Srb"].GetInt();
105  handle_dcch_pdu(
106  document, lcid, &rx_buf->at(rx_buf_offset), n - rx_buf_offset, ttcn3_helpers::get_follow_on_flag(document));
107  } else {
108  logger.error("Received unknown request.");
109  }
110 
111  return SRSRAN_SUCCESS;
112  }
