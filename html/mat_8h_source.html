135 /* Generic SIMD implementation for Minimum Mean Squared Error (MMSE) solver */
136 static inline void srsran_mat_2x2_mmse_csi_simd(simd_cf_t y0,
137  simd_cf_t y1,
138  simd_cf_t h00,
139  simd_cf_t h01,
140  simd_cf_t h10,
141  simd_cf_t h11,
142  simd_cf_t* x0,
143  simd_cf_t* x1,
144  simd_f_t* csi0,
145  simd_f_t* csi1,
146  float noise_estimate,
147  float norm)
148 {
149  simd_cf_t _noise_estimate;
150  simd_f_t _norm = srsran_simd_f_set1(norm);
151 
152 #if HAVE_NEON
153  _noise_estimate.val[0] = srsran_simd_f_set1(noise_estimate);
154  _noise_estimate.val[1] = srsran_simd_f_zero();
155 #else /* HAVE_NEON */
156  _noise_estimate.re = srsran_simd_f_set1(noise_estimate);
157  _noise_estimate.im = srsran_simd_f_zero();
158 #endif /* HAVE_NEON */
159 
160  /* 1. A = H' x H + No*/
161  simd_cf_t a00 = srsran_simd_cf_add(
162  srsran_simd_cf_add(srsran_simd_cf_conjprod(h00, h00), srsran_simd_cf_conjprod(h10, h10)), _noise_estimate);
163  simd_cf_t a01 = srsran_simd_cf_add(srsran_simd_cf_conjprod(h01, h00), srsran_simd_cf_conjprod(h11, h10));
164  simd_cf_t a10 = srsran_simd_cf_add(srsran_simd_cf_conjprod(h00, h01), srsran_simd_cf_conjprod(h10, h11));
165  simd_cf_t a11 = srsran_simd_cf_add(
166  srsran_simd_cf_add(srsran_simd_cf_conjprod(h01, h01), srsran_simd_cf_conjprod(h11, h11)), _noise_estimate);
167  simd_cf_t a_det_rcp = srsran_simd_cf_rcp(srsran_mat_2x2_det_simd(a00, a01, a10, a11));
168 
169  /* 2. B = inv(H' x H + No) = inv(A) */
170  simd_cf_t _norm2 = srsran_simd_cf_mul(a_det_rcp, _norm);
171  simd_cf_t b00 = srsran_simd_cf_prod(a11, _norm2);
172  simd_cf_t b01 = srsran_simd_cf_prod(srsran_simd_cf_neg(a01), _norm2);
173  simd_cf_t b10 = srsran_simd_cf_prod(srsran_simd_cf_neg(a10), _norm2);
174  simd_cf_t b11 = srsran_simd_cf_prod(a00, _norm2);
175 
176  /* 3. W = inv(H' x H + No) x H' = B x H' */
177  simd_cf_t w00 = srsran_simd_cf_add(srsran_simd_cf_conjprod(b00, h00), srsran_simd_cf_conjprod(b01, h01));
178  simd_cf_t w01 = srsran_simd_cf_add(srsran_simd_cf_conjprod(b00, h10), srsran_simd_cf_conjprod(b01, h11));
179  simd_cf_t w10 = srsran_simd_cf_add(srsran_simd_cf_conjprod(b10, h00), srsran_simd_cf_conjprod(b11, h01));
180  simd_cf_t w11 = srsran_simd_cf_add(srsran_simd_cf_conjprod(b10, h10), srsran_simd_cf_conjprod(b11, h11));
181 
182  /* 4. X = W x Y */
183  *x0 = srsran_simd_cf_add(srsran_simd_cf_prod(y0, w00), srsran_simd_cf_prod(y1, w01));
184  *x1 = srsran_simd_cf_add(srsran_simd_cf_prod(y0, w10), srsran_simd_cf_prod(y1, w11));
185 
186  /* 5. Extract CSI */
187  *csi0 = srsran_simd_f_rcp(srsran_simd_cf_re(b00));
188  *csi1 = srsran_simd_f_rcp(srsran_simd_cf_re(b11));
189 }
