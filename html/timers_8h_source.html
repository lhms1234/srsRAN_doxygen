235  explicit timer_handler(uint32_t capacity = 64)
236  {
237  time_wheel.resize(WHEEL_SIZE);
238  // Pre-reserve timers
239  while (timer_list.size() < capacity) {
240  timer_list.emplace_back(*this, timer_list.size());

243  for (auto it = timer_list.rbegin(); it != timer_list.rend(); ++it) {

249  void step_all()
250  {
251  std::unique_lock<std::mutex> lock(mutex);
252  uint32_t cur_time_local = cur_time.load(std::memory_order_relaxed) + 1;
253  auto& wheel_list = time_wheel[cur_time_local & WHEEL_MASK];
254 
255  for (auto it = wheel_list.begin(); it != wheel_list.end();) {
256  timer_impl& timer = timer_list[it->id];
257  ++it;
258  if (decode_timeout(timer.state.load(std::memory_order_relaxed)) == cur_time_local) {
259  // stop timer (callback has to see the timer has already expired)
260  stop_timer_(timer, true);
261 
262  // Call callback if configured
263  if (not timer.callback.is_empty()) {
264  // unlock mutex. It can happen that the callback tries to run a timer too
265  lock.unlock();
266 
267  timer.callback(timer.id);
268 
269  // Lock again to keep protecting the wheel
270  lock.lock();
271  }
272  }
273  }
274 
275  cur_time.fetch_add(1, std::memory_order_relaxed);
276  }

353  void start_run_(timer_impl& timer, uint32_t duration_ = 0)
354  {
355  uint64_t timer_old_state = timer.state.load(std::memory_order_relaxed);
356  duration_ = duration_ == 0 ? decode_duration(timer_old_state) : duration_;
357  uint32_t new_timeout = cur_time.load(std::memory_order_relaxed) + duration_;
358  size_t new_wheel_pos = new_timeout & WHEEL_MASK;
359 
360  uint32_t old_timeout = decode_timeout(timer_old_state);
361  bool was_running = decode_is_running(timer_old_state);
362  if (was_running and (old_timeout & WHEEL_MASK) == new_wheel_pos) {
363  // If no change in timer wheel position. Just update absolute timeout
364  timer.state.store(encode_state(RUNNING_FLAG, duration_, new_timeout), std::memory_order_relaxed);
365  return;
366  }
367 
368  // Stop timer if it was running, removing it from wheel in the process
369  if (was_running) {
370  time_wheel[old_timeout & WHEEL_MASK].pop(&timer);
371  nof_timers_running_--;
372  }
373 
374  // Insert timer in wheel
375  time_wheel[new_wheel_pos].push_front(&timer);
376  timer.state.store(encode_state(RUNNING_FLAG, duration_, new_timeout), std::memory_order_relaxed);
377  nof_timers_running_++;
378  }

381  void stop_timer_(timer_impl& timer, bool expiry)
382  {
383  uint64_t timer_old_state = timer.state.load(std::memory_order_relaxed);
384  if (not decode_is_running(timer_old_state)) {
385  return;
386  }
387 
388  // If already running, need to disconnect it from previous wheel
389  uint32_t old_timeout = decode_timeout(timer_old_state);
390  time_wheel[old_timeout & WHEEL_MASK].pop(&timer);
391  uint64_t new_state =
392  encode_state(expiry ? EXPIRED_FLAG : STOPPED_FLAG, decode_duration(timer_old_state), old_timeout);
393  timer.state.store(new_state, std::memory_order_relaxed);
394  nof_timers_running_--;
395  }
396 
397  std::atomic<tic_t> cur_time{0};
398  size_t nof_timers_running_ = 0, nof_free_timers = 0;
399  // using a deque to maintain reference validity on emplace_back. Also, this deque will only grow.
400  std::deque<timer_impl> timer_list;
401  srsran::intrusive_forward_list<timer_impl> free_list;
402  std::vector<srsran::intrusive_double_linked_list<timer_impl> > time_wheel;
