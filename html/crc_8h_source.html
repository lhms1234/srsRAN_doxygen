59  uint64_t crc = h->crcinit;

62  if (h->order > 8) {
63  // For more than 8 bits
64  uint32_t ord = h->order - 8U;
65  idx = ((crc >> (ord)) & 0xffU) ^ byte;
66  } else {
67  // For 8 bits or less
68  uint32_t ord = 8U - h->order;
69  idx = ((crc << (ord)) & 0xffU) ^ byte;
70  }
71 
72  crc = (crc << 8U) ^ h->table[idx];
