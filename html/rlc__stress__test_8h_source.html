40 inline int write_pdu_to_pcap(const std::unique_ptr<srsran::mac_pcap>& pcap_handle,
41  const bool is_dl,
42  const uint32_t lcid,
43  const uint8_t* payload,
44  const uint32_t len)
45 {
46  const uint32_t PCAP_CRNTI = 0x1001;
47  const uint32_t PCAP_TTI = 666;
48  if (pcap_handle) {
49  srsran::byte_buffer_t tx_buffer;
50  srsran::mac_sch_pdu_nr tx_pdu;
51  tx_pdu.init_tx(&tx_buffer, len + 10);
52  tx_pdu.add_sdu(lcid, payload, len);
53  tx_pdu.pack();
54  if (is_dl) {
55  pcap_handle->write_dl_crnti_nr(tx_buffer.msg, tx_buffer.N_bytes, PCAP_CRNTI, true, PCAP_TTI);
56  } else {
57  pcap_handle->write_ul_crnti_nr(tx_buffer.msg, tx_buffer.N_bytes, PCAP_CRNTI, true, PCAP_TTI);
58  }
59 
60  return SRSRAN_SUCCESS;
61  }
62  return SRSRAN_ERROR;
