314 private:
315  bool round_robin_pop_(myobj* value)
316  {
317  // Round-robin for all queues
318  auto q_it = queues.begin() + spin_idx;
319  uint32_t count = 0;
320  for (; count < queues.size(); ++count, ++q_it) {
321  if (q_it == queues.end()) {
322  q_it = queues.begin(); // wrap-around
323  }
324  bool try_lock_success = true;
325  if (q_it->try_pop(*value, try_lock_success)) {
326  spin_idx = (spin_idx + count + 1) % queues.size();
327  return true;
328  }
329  if (not try_lock_success) {
330  // restart RR search, as there was a collision with a producer
331  count = 0;
332  }
333  }
334  return false;
335  }
