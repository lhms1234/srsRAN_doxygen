225  float get_speed_kmph(uint32_t tti)
226  {

229  // Advance read pointer for old TTI
230  while (read_idx != write_idx and TTI_SUB(tti, speed_data[read_idx].tti) > MAX_AGE_SPEED_VALUES) {
231  read_idx = (read_idx + 1) % MAX_NOF_SPEED_VALUES;
232 
233  // If there us no data, make last_tti invalid to prevent invalid TTI difference
234  if (read_idx == write_idx) {
235  last_tti = -1;
236  }
237  }
238 
239  // Early return if there is not enough data to calculate speed
240  uint32_t nof_values = ((write_idx + MAX_NOF_SPEED_VALUES) - read_idx) % MAX_NOF_SPEED_VALUES;
241  if (nof_values < MIN_NOF_SPEED_VALUES) {
242  return 0.0f;
243  }
244 
245  // Compute speed from gathered data
246  float sum_t = 0.0f;
247  float sum_d = 0.0f;
248  for (uint32_t i = read_idx; i != write_idx; i = (i + 1) % MAX_NOF_SPEED_VALUES) {
249  sum_t += speed_data[i].delta_t;
250  sum_d += speed_data[i].delta_d;
251  }
252  if (!std::isnormal(sum_t)) {
253  return 0.0f; // Avoid zero division
254  }
255  float speed_mps = sum_d / sum_t; // Speed in m/s
