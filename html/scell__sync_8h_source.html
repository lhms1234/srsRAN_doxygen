76  void run_state_search_pss(uint32_t tti, cf_t* buffer)
77  {
78  uint32_t peak_pos = 0;
79 
80  // Append new base-band
81  if (buffer == nullptr) {
82  srsran_vec_cf_zero(&temp[sf_len], sf_len);
83  } else {
84  srsran_vec_cf_copy(&temp[sf_len], buffer, sf_len);
85  }
86 
87  // Run PSS search
88  switch (srsran_sync_find(&find_pss, temp.data(), 0, &peak_pos)) {
89  case SRSRAN_SYNC_FOUND:
90  if (callback != nullptr) {
91  // Calculate Sample Offset from TTI difference
92  int tti_mod = (int)tti % (SRSRAN_NOF_SF_X_FRAME / 2);
93  int tti_offset = (tti_mod < 3) ? tti_mod : (tti_mod - SRSRAN_NOF_SF_X_FRAME / 2);
94 
95  // Calculate sample offset from PSS correlation peak
96  int offset = (int)(peak_pos - (3 * sf_len) / 2);
97 
98  // Provide offset through feedback interface
99  callback->set_rx_channel_offset(channel, offset + tti_offset * sf_len);
100  }
101  state = STATE_IN_SYNCH;
102  break;
103  case SRSRAN_SYNC_FOUND_NOSPACE:
104  ERROR("No space error");
105  break;
106  case SRSRAN_SYNC_NOFOUND:
107  // Ignore
108  break;
109  case SRSRAN_SYNC_ERROR:
110  ERROR("Error finding PSS");
111  break;
112  }
113 
114  // If the state has not changed, copy new data into the temp buffer
115  if (state == STATE_SEARCH_PSS) {
116  srsran_vec_cf_copy(&temp[0], buffer, sf_len);
117  }
118  }

136  // Resizes synchronization object. As the secondary serving cell base-band might be unaligned respect the primary

139  // subframes is a simple and conservative buffer size.
