208  bool run_tti()
209  {
210  // Get gNb worker
211  srsenb::nr::slot_worker* gnb_worker = gnb_phy.wait_worker(slot_idx);
212  if (gnb_worker == nullptr) {
213  return false;
214  }
215 
216  // Feed gNb the UE transmitted signal
217  std::vector<cf_t*> gnb_rx_buffers(1);
218  gnb_rx_buffers[0] = gnb_worker->get_buffer_rx(0);
219  ue_radio.read_tx(gnb_rx_buffers.data(), sf_sz);
220 
221  // Run the UL channel simulator
222  ul_channel.run(gnb_rx_buffers.data(), gnb_rx_buffers.data(), (uint32_t)sf_sz, gnb_rx_time.get(0));
223 
224  // Set gNb TX time
225  srsran::rf_timestamp_t gnb_time = gnb_rx_time;
226  gnb_time.add(TX_ENB_DELAY * 1e-3);
227 
228  // Advance gNb Rx time
229  gnb_rx_time.add(1e-3);
230 
231  // Set gNb context
232  srsran::phy_common_interface::worker_context_t gnb_context;
233  gnb_context.sf_idx = slot_idx;
234  gnb_context.worker_ptr = gnb_worker;
235  gnb_context.last = true; // Set last if standalone
236  gnb_context.tx_time.copy(gnb_time);
237  gnb_worker->set_context(gnb_context);
238 
239  // Start gNb work
240  gnb_phy_com.push_semaphore(gnb_worker);
241  gnb_phy.start_worker(gnb_worker);
242 
243  // Feed UE the gNb transmitted signal
244  srsran::rf_timestamp_t ue_time = {};
245  std::vector<cf_t*> ue_rx_buffers(1);
246  ue_rx_buffers[0] = rf_buffer.get(0);
247  gnb_phy_com.read(ue_rx_buffers, sf_sz, ue_time);
248 
249  // Run the DL channel simulator
250  dl_channel.run(ue_rx_buffers.data(), ue_rx_buffers.data(), (uint32_t)sf_sz, ue_time.get(0));
251 
252  // Write signal in UE radio buffer, this triggers UE to work
253  ue_radio.write_rx(ue_rx_buffers.data(), sf_sz);
254 
255  // Throttle UE PHY by running stack tick
256  ue_stack.tick();
257 
258  // Increment slot index, the slot index shall be continuous
259  slot_idx = (slot_idx + 1) % (1024 * SRSRAN_NSLOTS_PER_FRAME_NR(srsran_subcarrier_spacing_15kHz));
260 
261  // Increment slot counter and determine end of execution
262  slot_count++;
263  return slot_count <= duration_slots;
264  }
